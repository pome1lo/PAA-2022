uint function factorial(uint x)
{
	new uint ret = 1;
	if (x < 0)
	{
		ret =  0;
	}
	if (x == 0)
	{
		ret = 1;
	}
	if (x > 0)
	{
		new uint y = x - 1;
		ret = factorial(y) * x;
	}
	return ret;
}
